

---DROP TABLE  BU_CAPTOOLS_WORK.ANL_SFICS_JUN25;

CREATE TABLE BU_CAPTOOLS_WORK.ANL_SFICS_JUN25 AS 
SELECT *--DISTINCT F1.nome_ctgr_sfics,F3.activity_template 
   -- F1.nome_ctgr_sfics,F2.SFCIS
    ,CASE
        WHEN F2.CHAVE_SB IS NOT NULL THEN 'Tático DEZ24 - Cruzamento SB'
        ELSE 'Tático DEZ24 - Sem Cruzamento SB'
    END AS FLAG_CRUZ_SB
    ,CASE 
        WHEN TRIM(F1.nome_ctgr_sfics)='Nao Aplicavel' AND TRIM(F2.SFCIS)='' THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics)='Nao Aplicavel' AND TRIM(F2.SFCIS)<>'' THEN 'Sem SFICS no MV'
        WHEN TRIM(F2.SFCIS)='' THEN 'Sem preenchimento SB'
        WHEN substr(TRIM(F1.nome_ctgr_sfics),1,length(TRIM(F1.nome_ctgr_sfics))-1)=TRIM(F2.SFCIS) THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics)='A.3.1.' AND TRIM(F2.SFCIS) IN ('A.3.1.2.1','A.3.1.2.2') THEN 'OK' 
        WHEN TRIM(F1.nome_ctgr_sfics)='A.3.7.' AND TRIM(F2.SFCIS) IN ('A.3.7.2.1','A.3.7.2.2','A.3.7.5.2') THEN 'OK'
        WHEN UPPER(TRIM(F1.nome_ctgr_sfics)) LIKE 'PURE%' AND UPPER(TRIM(F2.SFCIS)) LIKE 'PURE%' THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) IN ('Sports and cultural education centres','Student loans') AND TRIM(F2.SFCIS)='B.1' THEN 'OK'
        WHEN CHAVE_SB IS NULL THEN 'Sem cruzamento'
        ELSE 'NOK'
    END AS FLAG_SB_SFICS
    ,CASE
        WHEN F3.CHAVE_CDG IS NOT NULL THEN 'Tático DEZ24 - Cruzamento CDG'
        ELSE 'Tático DEZ24 - Sem Cruzamento CDG'
    END AS FLAG_CRUZ_CDG
    ,CASE 
        WHEN TRIM(F1.nome_ctgr_sfics) LIKE 'A.1.%' AND TRIM(F3.activity_template)='AGFGR001' THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) LIKE 'A.2.%' AND TRIM(F3.activity_template) IN ('AGFGR003','AGFGR002') THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) LIKE 'A.7.%' AND TRIM(F3.activity_template)='AGFGR005' THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) LIKE 'A.3.%' AND TRIM(F3.activity_template) IN ('AGFGR008','AGFGR007','AGFGR006') THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) LIKE 'A.6.%' AND TRIM(F3.activity_template)='AGFGR009' THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) LIKE 'A.8.%' AND TRIM(F3.activity_template)='AGFGR010' THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) IN ('KPI Linked - Green','Sustainability Linked Finance - Default') AND TRIM(F3.activity_template)='AGFSL001' THEN 'OK'
        WHEN TRIM(F1.nome_ctgr_sfics) IN ( 
                                            'Building of healthcare facilities',
                                            'Financing with preferential financial or payment terms to entities and individuals that have been impacted by natural, armed, health and/or human-made disasters, as well as severe socioeconomic situations.',
                                            'Health services',
                                            'Lending to non-profit organizations and charities that meet Banco Santanders guidelines and advance the green and social themes',
                                            'Research and develpment (R&D), pharmaceutical and medical manufacturing',
                                            'Sports and cultural education centres',
                                            'Student loans'
                                        ) AND TRIM(F3.activity_template)='AGFSO001' THEN 'OK'
        WHEN F3.CHAVE_CDG IS NULL THEN 'Sem cruzamento'
        WHEN TRIM(F1.nome_ctgr_sfics)='Nao Aplicavel' and F3.activity_template IS NOT NULL THEN 'Sem SFICS no MV'
        WHEN TRIM(F1.nome_ctgr_sfics)='Pure Green counterparty - Default' and F3.activity_template IS NOT NULL THEN 'NOK - Pure Green vs Categoria SFICS'  
        ELSE 'NOK'
    END AS FLAG_CDG_SFICS
    ,CASE 
        WHEN F2.CHAVE_SB IS NULL AND F3.CHAVE_CDG IS NULL AND TRIM(F1.nome_ctgr_sfics)='Nao Aplicavel' THEN 'Sem cruzamento & Sem SFICS'
        WHEN F2.CHAVE_SB IS NULL AND F3.CHAVE_CDG IS NULL AND TRIM(F1.nome_ctgr_sfics)<>'Nao Aplicavel' THEN 'Sem cruzamento & Com SFICS'
        ELSE 'Com cruzamento'
    END AS CRUZAMENTO 
        
        
FROM 
    (
    SELECT concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE, nome_ctgr_sfics 
    FROM bu_esg_work.modesg_out_sfics_taxon_europeia
    WHERE ref_date='2025-06-30'
    )F1
LEFT JOIN 
    (
    SELECT concat(cempresa_ct,cbalcao_ct,cnumecta_ct,zdeposit_ct) AS CHAVE_SB,SFCIS
    FROM bu_esg_work.mapeamento_sfics_dez24_v2
    )F2 ON F1.CHAVE=F2.CHAVE_SB
LEFT JOIN 
    (
    SELECT DISTINCT concat(cempresa_ct,cbalcao_ct,cnumecta_ct,zdeposit_ct) AS CHAVE_CDG, activity_template
    FROM bu_esg_work.rf_pilar3_cdg_tabaux1_Dez24_v2
    )F3 ON F1.CHAVE=F3.CHAVE_CDG
-- HAVING FLAG_SB_SFICS='NOK'
;

SELECT flag_cruz_sb,flag_sb_sfics,flag_cruz_cdg,flag_cdg_sfics,cruzamento,SUM(VALOR_BRUTO) AS AMOUNT, count(*) AS NUM_REG

FROM 
    (
    SELECT DISTINCT *
    FROM BU_CAPTOOLS_WORK.ANL_SFICS_JUN25
    )A1
LEFT JOIN 
    (
    SELECT F1.CHAVE_MASTER,SUM(F2.SALDO) AS VALOR_BRUTO
    FROM 
        (
        SELECT concat(cempresa_fr,cbalcao_fr,cnumecta_fr,zdeposit_fr) AS CHAVE_FR,concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_MASTER
        FROM cd_captools.kt_chaves_finrep
        WHERE ref_date='2025-06-30'
        )F1
    INNER JOIN
        (
        SELECT concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_FINREP,SUM(valor_bruto_on_a) AS SALDO
        FROM cd_captools.fr004_cto
        WHERE ref_date='2025-06-30'
        GROUP BY 1
        )F2 ON F1.CHAVE_FR=F2.CHAVE_FINREP
    GROUP BY 1 
    )A2 ON A1.CHAVE=A2.CHAVE_MASTER
GROUP BY 1,2,3,4,5
;

-- Casos enviados sheet 2.1.
SELECT DISTINCT A1.*,A2.CHAVE_207,cempcta,ckbalcao,cknumcta,zdeposit_mis,CKMETAMIS,CKPRODMI,CMETANSEG,purpose_code,CHAVE_BEM_12,combustivel,TIPO_EQP,CHAVE_BEM_8,EPC,quality_score,a3.VALOR_BRUTO
FROM 
    (
    SELECT *
    FROM BU_CAPTOOLS_WORK.ANL_SFICS_JUN25
    WHERE flag_cruz_sb='Tático DEZ24 - Sem Cruzamento SB'
        AND flag_sb_sfics='Sem cruzamento'
        AND flag_cruz_cdg='Tático DEZ24 - Cruzamento CDG'
        AND flag_cdg_sfics='Sem SFICS no MV'
        AND cruzamento='Com cruzamento'
    )A1
LEFT JOIN 
    (
    SELECT *
    FROM BU_CAPTOOLS_WORK.MODESG_OUT_SFICS_TAXON_EUROPEIA_BASE_AUX_JUN25
    -- WHERE CHAVE_CTO='31000317867846096000000000000000'
    )A2 ON A1.CHAVE=A2.CHAVE_CTO
-- WHERE A2.CHAVE_CTO IS NULL 
LEFT JOIN 
    (
    SELECT F1.CHAVE_MASTER,SUM(F2.SALDO) AS VALOR_BRUTO
    FROM 
        (
        SELECT concat(cempresa_fr,cbalcao_fr,cnumecta_fr,zdeposit_fr) AS CHAVE_FR,concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_MASTER
        FROM cd_captools.kt_chaves_finrep
        WHERE ref_date='2025-06-30'
        )F1
    INNER JOIN
        (
        SELECT concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_FINREP,SUM(valor_bruto_on_a) AS SALDO
        FROM cd_captools.fr004_cto
        WHERE ref_date='2025-06-30'
        GROUP BY 1
        )F2 ON F1.CHAVE_FR=F2.CHAVE_FINREP
    GROUP BY 1 
    )A3 ON A1.CHAVE=A3.CHAVE_MASTER
LEFT JOIN 
    (
    SELECT concat(cempcta,ckbalcao,cknumcta,zdeposit_mis) AS CHAVE_MIS,concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_MASTER,cempcta,ckbalcao,cknumcta,zdeposit_mis
    FROM cd_captools.kt_chaves_mis
    WHERE ref_date='2025-06-30'
    )A7 ON A1.CHAVE=A7.CHAVE_MASTER AND A2.CHAVE_207=A7.CHAVE_MIS
ORDER BY CHAVE
;
-- casos enviados sheet a.2.

SELECT DISTINCT A1.*,ZCLIENTE,A2.CHAVE_207,cempcta,ckbalcao,cknumcta,zdeposit_mis,CKMETAMIS,CKPRODMI,CMETANSEG,purpose_code,CHAVE_BEM_12,combustivel,TIPO_EQP,CHAVE_BEM_8,EPC,quality_score,a3.VALOR_BRUTO
FROM 
    (
    SELECT *
    FROM BU_CAPTOOLS_WORK.ANL_SFICS_JUN25
    -- WHERE flag_cruz_sb='Tático DEZ24 - Sem Cruzamento SB'
    --     AND flag_sb_sfics='Sem cruzamento'
    --     AND flag_cruz_cdg='Tático DEZ24 - Cruzamento CDG'
    --     AND flag_cdg_sfics='Sem SFICS no MV'
    --     AND cruzamento='Com cruzamento'
    )A1
LEFT JOIN 
    (
    SELECT *
    FROM BU_CAPTOOLS_WORK.MODESG_OUT_SFICS_TAXON_EUROPEIA_BASE_AUX_JUN25
    -- WHERE CHAVE_CTO='31000317867846096000000000000000'
    )A2 ON A1.CHAVE=A2.CHAVE_CTO
-- WHERE A2.CHAVE_CTO IS NULL 
LEFT JOIN 
    (
    SELECT F1.CHAVE_MASTER,SUM(F2.SALDO) AS VALOR_BRUTO
    FROM 
        (
        SELECT concat(cempresa_fr,cbalcao_fr,cnumecta_fr,zdeposit_fr) AS CHAVE_FR,concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_MASTER
        FROM cd_captools.kt_chaves_finrep
        WHERE ref_date='2025-06-30'
        )F1
    INNER JOIN
        (
        SELECT concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_FINREP,SUM(valor_bruto_on_a) AS SALDO
        FROM cd_captools.fr004_cto
        WHERE ref_date='2025-06-30'
        GROUP BY 1
        )F2 ON F1.CHAVE_FR=F2.CHAVE_FINREP
    GROUP BY 1 
    )A3 ON A1.CHAVE=A3.CHAVE_MASTER
LEFT JOIN 
    (
    SELECT concat(cempcta,ckbalcao,cknumcta,zdeposit_mis) AS CHAVE_MIS,concat(cempresa,cbalcao,cnumecta,zdeposit) AS CHAVE_MASTER,cempcta,ckbalcao,cknumcta,zdeposit_mis
    FROM cd_captools.kt_chaves_mis
    WHERE ref_date='2025-06-30'
    )A7 ON A1.CHAVE=A7.CHAVE_MASTER AND A2.CHAVE_207=A7.CHAVE_MIS

WHERE CONCAT(flag_cruz_sb,flag_sb_sfics,flag_cruz_cdg,flag_cdg_sfics,cruzamento)
IN 
(
'Tático DEZ24 - Sem Cruzamento SBSem cruzamentoTático DEZ24 - Cruzamento CDGNOK - Pure Green vs Categoria SFICSCom cruzamento'
,'Tático DEZ24 - Sem Cruzamento SBSem cruzamentoTático DEZ24 - Cruzamento CDGNOKCom cruzamento'
)


ORDER BY CHAVE
;
