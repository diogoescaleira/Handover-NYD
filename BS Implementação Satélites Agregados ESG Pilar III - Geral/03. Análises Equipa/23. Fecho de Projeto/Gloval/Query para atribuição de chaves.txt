WITH AA AS(
SELECT
    A.CHAVE_GLOVAL_JUN25, 
    B.property_bank_id, 
    B.property_branch_code, 
    B.property_contract_id, 
    B.property_reference_code,
    A.FIABILIDAD_JUN25,
    A.clase_energetica_JUN25, 
    A.emisiones_JUN25, 
    A.consumos_JUN25, 
    rank() OVER(ORDER BY chave_gloval, property_bank_id, property_branch_code, property_contract_id, property_reference_code, FIABILIDAD_JUN25, clase_energetica_JUN25, emisiones_JUN25, consumos_JUN25 DESC) as ranking
FROM
    (
    SELECT
        CHAVE_GLOVAL AS CHAVE_GLOVAL_JUN25, 
        FIABILIDAD AS FIABILIDAD_JUN25, 
        clase_energetica AS clase_energetica_JUN25, 
        emisiones AS emisiones_JUN25, 
        consumos AS consumos_JUN25
    FROM bu_ESG_WORK.gloval_atualizacao_info_jun25
    ) A
    
    LEFT JOIN
    
    (
    SELECT DISTINCT
        chave_gloval, 
        property_bank_id, 
        property_branch_code, 
        property_contract_id, 
        property_reference_code
    FROM cd_esg.energy_certificate_est_input
    ) B
    
    ON A.CHAVE_GLOVAL_JUN25=B.chave_gloval
)
SELECT *
FROM AA
WHERE RANKING >= ${Ini_extracao}
and RANKING < ${Fim_extracao}
;
