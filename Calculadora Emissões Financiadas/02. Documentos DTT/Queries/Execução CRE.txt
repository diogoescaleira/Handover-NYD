 ${ref_date 2024-06-30};
 ${descri Execucao padrao};

INSERT INTO TABLE bu_esg_work.CRE_Emissions
SELECT '${ref_date}' as ref_date,
        from_unixtime(unix_timestamp()) as execution_date,
		case when lid.max_load_id is null then 1
	    else lid.max_load_id
	    END as load_id,
        '${descri}' as description,
        p.cempresa_ct,
        p.cbalcao_ct,
        p.cnumecta_ct,
        p.zdeposit_ct,
        p.ckbalbem,
        p.ckctabem,
        p.ckrefbem,
        p.area_util,
        p.Certeficado_Energetico as ienerg,
        p.SCORE_ENRG as score_pcaf,
        p.Data_Vencimento_Contrato,
        p.Tipo_Imovel_Desc,
        p.Tipo_Imovel_Desc_PCAF,
        p.Outstanding_Amount,
        p.Value_At_Origination,
        p.ATRIBUTION_FACTOR,
        p.Calculated_Emissions_m2,
        p.Calculated_emissions_building,
        p.Calculated_emissions_building * p.ATRIBUTION_FACTOR AS Financed_Emissions
FROM
  (SELECT 1 as dummy_id,
          t.cempresa_ct,
          t.cbalcao_ct,
          t.cnumecta_ct,
          t.zdeposit_ct,
          t.ckbalbem,
          t.ckctabem,
          t.ckrefbem,
          t.Data_Vencimento_Contrato,
          t.Certeficado_Energetico,
          t.Tipo_Imovel_Code,
          t.Tipo_Imovel_Desc,
          t.Tipo_Imovel_Desc_PCAF,
          t.Outstanding_Amount,
          t.Value_At_Origination,
          t.area_util,
          t.SCORE_ENRG,
          CASE
              WHEN t.SCORE_ENRG = '3a' THEN e3a.avg_emissions
              WHEN t.SCORE_ENRG = '3' THEN NULL
              WHEN t.SCORE_ENRG = '4' THEN e4.avg_emissions
              WHEN t.SCORE_ENRG = '5' THEN NULL
              ELSE NULL
          END AS Calculated_Emissions_m2,
          CASE
              WHEN t.SCORE_ENRG = '3a' THEN e3a.avg_emissions * t.area_util
              WHEN t.SCORE_ENRG = '3' THEN e3.avg_emissions_building
              WHEN t.SCORE_ENRG = '4' THEN e4.avg_emissions * t.area_util
              WHEN t.SCORE_ENRG = '5' THEN e5.avg_emissions_building
          END AS Calculated_emissions_building,
          CASE
              WHEN t.Outstanding_Amount > t.Value_At_Origination THEN 1
              ELSE t.Outstanding_Amount / t.Value_At_Origination
          END AS ATRIBUTION_FACTOR
   FROM
     (SELECT x.cempresa_ct,
             x.cbalcao_ct,
             x.cnumecta_ct,
             x.zdeposit_ct,
             x.ckbalbem,
             x.ckctabem,
             x.ckrefbem,
             ct004.ddvencim AS Data_Vencimento_Contrato,
             b.net_floor_area AS Area_Util,
             CASE
                 WHEN a.epc = 'G' THEN 'F'
                 ELSE a.epc
             END AS Certeficado_Energetico,
             gt018.cgarant_bem AS Tipo_Imovel_Code,
             tat91.tayd91c0_gelemtab AS Tipo_Imovel_Desc,
             CASE
                 WHEN tat91.tayd91c0_gelemtab = 'IMOVEL P/ ESCRITORIO'
                      OR tat91.tayd91c0_gelemtab = 'IMOVEL P/ ESCRITORIO - EM CONSTRUCAO' THEN 'Office'
                 WHEN tat91.tayd91c0_gelemtab = 'IMOVEL P/ COMERCIO'
                      OR tat91.tayd91c0_gelemtab = 'IMOVEL P/ COMERCIO - EM CONSTRUCAO' THEN 'Retail'
                 WHEN tat91.tayd91c0_gelemtab = 'IMOVEL P/ HOTEL' THEN 'Retail'
                 WHEN tat91.tayd91c0_gelemtab = 'IMOVEL P/ SAUDE' THEN 'Healthcare'
                 WHEN tat91.tayd91c0_gelemtab = 'IMOVEL P/ ARMAZEM'
                      OR tat91.tayd91c0_gelemtab= 'IMOVEL P/ ARMAZEM - EM CONSTRUCAO' THEN 'Distribution warehouse'
                 WHEN tat91.tayd91c0_gelemtab = 'IMOVEL P/ INDUSTRIA'
                      OR tat91.tayd91c0_gelemtab = 'IMOVEL P/ EDUCACAO'
                      OR tat91.tayd91c0_gelemtab = 'IMOVEL OU FRACAO P/ ESTACIONAMENTO LIGADOS A EXPLORACAO'
                      OR tat91.tayd91c0_gelemtab = 'IMOVEIS COM FINALIDADE INDEFINIDA' THEN 'Non-residential total'
                 ELSE 'Non-residential total'
             END AS Tipo_Imovel_Desc_PCAF,
             fr004.capital_juro_on_a *(-1) * gt701_new.mavaliai_resp_bem /sum(gt701_new.mavaliai_resp_bem) OVER (PARTITION BY concat(x.cempresa_ct, x.cbalcao_ct, x.cnumecta_ct, x.zdeposit_ct)) AS Outstanding_Amount,
             gt701_new.mavaliai_resp_bem AS Value_At_Origination,
             CASE
                 WHEN (a.epc IS NOT NULL
                       AND a.epc != '0'
                       AND b.net_floor_area IS NOT NULL
                       AND b.net_floor_area != 0) THEN '3a'
                 WHEN (a.epc IS NOT NULL
                       AND a.epc != '0') THEN '3'
                 WHEN (b.net_floor_area IS NOT NULL
                       AND b.net_floor_area != 0) THEN '4'
                 WHEN (b.net_floor_area IS NULL
                       OR b.net_floor_area = 0) THEN '5'
                 ELSE 'ERROR'
             END AS SCORE_ENRG
      FROM bu_esg_work.local_pilar3_temp1_emi_tabaux18_jun24 AS x
      LEFT JOIN
        (SELECT DISTINCT f.cempresa_ct,
                         f.cbalcao_ct,
                         f.cnumecta_ct,
                         f.zdeposit_ct,
                         f.ckbalbem,
                         f.ckctabem,
                         f.ckrefbem,
                         f.`32_type_collateral`
         FROM bu_esg_work.local_pilar3_temp1_emi_tabaux5_jun24 AS f) AS xx ON x.cempresa_ct = xx.cempresa_ct
      AND x.cbalcao_ct = xx.cbalcao_ct
      AND x.cnumecta_ct = xx.cnumecta_ct
      AND x.zdeposit_ct = xx.zdeposit_ct
      AND x.ckbalbem = xx.ckbalbem
      AND x.ckctabem = xx.ckctabem
      AND x.ckrefbem = xx.ckrefbem
      LEFT JOIN
        (SELECT ct.cbalcao,
                ct.cnumecta,
                ct.zdeposit,
                ct.cempresa,
                ct.ddvencim
         FROM cd_captools.ct004_univ_cto AS ct
         WHERE ref_date ='${ref_date}') AS ct004 ON x.cbalcao_ct = ct004.cbalcao
      AND x.cnumecta_ct = ct004.cnumecta
      AND x.zdeposit_ct = ct004.zdeposit
      AND x.cempresa_ct = ct004.cempresa
      LEFT JOIN
        (SELECT b1.*
         FROM business_assets.property AS b1
         WHERE data_date_part =
             ( SELECT max(b2.data_date_part)
              FROM business_assets.property_energy_certificate b2
              WHERE b1.property_branch_code = b2.property_branch_code
                AND b1.property_contract_id = b2.property_contract_id
                AND b1.property_reference_code = b2.property_reference_code )) AS b ON x.ckbalbem = b.property_branch_code
      AND x.ckctabem = b.property_contract_id
      AND x.ckrefbem = b.property_reference_code
      LEFT JOIN
        (SELECT a1.*
         FROM business_assets.property_energy_certificate AS a1
         WHERE data_date_part =
             ( SELECT max(a2.data_date_part)
              FROM business_assets.property_energy_certificate a2
              WHERE a1.property_branch_code = a2.property_branch_code
                AND a1.property_contract_id = a2.property_contract_id
                AND a1.property_reference_code = a2.property_reference_code )) AS a ON x.ckbalbem = a.property_branch_code
      AND x.ckctabem = a.property_contract_id
      AND x.ckrefbem = a.property_reference_code
      LEFT JOIN
        (SELECT gt.*
         FROM cd_garantias.gt018_rateio_aval AS gt
         WHERE gt.ref_date = '${ref_date}'
           AND gt.tipologia = 'IMO') AS gt018 ON x.cempresa_ct = gt018.cempresp
      AND x.cbalcao_ct = gt018.ckbalres
      AND x.cnumecta_ct = gt018.ckctares
      AND x.zdeposit_ct = gt018.ckrefresp
      AND x.ckbalbem = gt018.ckbalbem
      AND x.ckctabem = gt018.ckctabem
      AND x.ckrefbem = gt018.ckrefbem
      LEFT JOIN
        (SELECT tat.*
         FROM cd_estruturais.tat91_tabelas AS tat
         WHERE tat.tayd91c0_ctabela = '236'
           AND tat.data_date_part =
             ( SELECT MAX(tat91_1.data_date_part)
              FROM cd_estruturais.tat91_tabelas tat91_1
              WHERE tat91_1.tayd91c0_celemtab = tat.tayd91c0_celemtab)) AS tat91 ON gt018.cgarant_bem = tat91.tayd91c0_celemtab
      LEFT JOIN
        (SELECT tra.cempresa,
                tra.cbalcao,
                tra.cnumecta,
                tra.zdeposit,
                tra.cempresa_fr,
                tra.cbalcao_fr,
                tra.cnumecta_fr,
                tra.zdeposit_fr
         FROM cd_captools.kt_chaves_finrep AS tra
         WHERE tra.ref_date ='${ref_date}') AS tradutor ON x.cempresa_ct = tradutor.cempresa
      AND x.cbalcao_ct = tradutor.cbalcao
      AND x.cnumecta_ct = tradutor.cnumecta
      AND x.zdeposit_ct = tradutor.zdeposit
      LEFT JOIN
        (SELECT fr.cempresa,
                fr.cbalcao,
                fr.cnumecta,
                fr.zdeposit,
                fr.capital_juro_on_a
         FROM cd_captools.fr004_cto AS fr
         WHERE fr.ref_date ='${ref_date}') AS fr004 ON tradutor.cempresa_fr = fr004.cempresa
      AND tradutor.cbalcao_fr = fr004.cbalcao
      AND tradutor.cnumecta_fr = fr004.cnumecta
      AND tradutor.zdeposit_fr = fr004.zdeposit
      LEFT JOIN
        (SELECT f1.cempbem,
                f1.ckbalbem,
                f1.ckctabem,
                f1.ckrefbem,
                f1.dtvaliai_resp_bem,
                SUM(f1.mavaliai_resp_bem) AS mavaliai_resp_bem
         FROM
           (SELECT univ_gt018.*,
                   GREATEST(gt001.dini_rel, gt008.dini_rel) AS dini_rel_resp_bem,
                   MIN(GREATEST(gt001.dini_rel, gt008.dini_rel)) OVER (PARTITION BY univ_gt018.cempbem, univ_gt018.ckbalbem, univ_gt018.ckctabem, univ_gt018.ckrefbem) AS dini_rel_resp_bem_min
            FROM
              (SELECT sub_a.*
               FROM
                 (SELECT gt701.*,
                         MIN(dtvaliai_resp_bem) OVER (PARTITION BY cempbem, ckbalbem, ckctabem, ckrefbem) AS min_data_bem
                  FROM cd_garantias.gt701_hist_mavaliai AS gt701
                  WHERE gt701.tipologia = 'IMO'
                    AND gt701.flag_ativo = 1
                    AND concat(gt701.cempresp, gt701.ckctares, gt701.ckbalres, gt701.ckrefresp, gt701.cempresa, gt701.ckbalcao, gt701.cknumcta, gt701.comp_id_gar, gt701.cempbem, gt701.ckbalbem, gt701.ckctabem, gt701.ckrefbem) IN
                      (SELECT CONCAT(cempresp, ckctares, ckbalres, ckrefresp, cempresa, ckbalcao, cknumcta, comp_id_gar, cempbem, ckbalbem, ckctabem, ckrefbem)
                       FROM cd_garantias.gt018_rateio_aval
                       WHERE ref_date = '${ref_date}'
                         AND tipologia = 'IMO' )) AS sub_a
               WHERE sub_a.dtvaliai_resp_bem = sub_a.min_data_bem) AS univ_gt018
            LEFT JOIN
              (SELECT gt1.*
               FROM cd_garantias.gt001_trz_rsp_gt AS gt1
               WHERE gt1.ref_date = '${ref_date}') AS gt001 ON univ_gt018.cempresp = gt001.cempresp
            AND univ_gt018.ckctares = gt001.ckctares
            AND univ_gt018.ckbalres = gt001.ckbalres
            AND univ_gt018.ckrefresp = gt001.ckrefresp
            AND univ_gt018.cempresa = gt001.cempresa
            AND univ_gt018.ckbalcao = gt001.ckbalcao
            AND univ_gt018.cknumcta = gt001.cknumcta
            AND univ_gt018.comp_id_gar = gt001.comp_id_gar
            LEFT JOIN
              (SELECT gt8.*
               FROM cd_garantias.gt008_trz_gt_imv AS gt8
               WHERE gt8.ref_date = '${ref_date}') AS gt008 ON univ_gt018.cempbem = gt008.cempbem
            AND univ_gt018.ckctabem = gt008.ckctabem
            AND univ_gt018.ckbalbem = gt008.ckbalbem
            AND univ_gt018.ckrefbem = gt008.ckrefbem
            AND univ_gt018.cempresa = gt008.cempresa
            AND univ_gt018.ckbalcao = gt008.ckbalcao
            AND univ_gt018.cknumcta = gt008.cknumcta
            AND univ_gt018.comp_id_gar = gt008.comp_id_gar) AS f1
         WHERE f1.dini_rel_resp_bem = f1.dini_rel_resp_bem_min
         GROUP BY f1.cempbem,
                  f1.ckbalbem,
                  f1.ckctabem,
                  f1.ckrefbem,
                  f1.dtvaliai_resp_bem) AS gt701_new ON x.ckbalbem = gt701_new.ckbalbem
      AND x.ckctabem = gt701_new.ckctabem
      AND x.ckrefbem = gt701_new.ckrefbem
      WHERE x.calculation_approach_2 = 'Real Estate'
        AND xx.`32_type_collateral` = 'COMERCIAL') AS t
   LEFT JOIN
     (SELECT sube3a.epc,
             sube3a.Tipo_Finalidade_Bem_fixed,
             AVG(sube3a.scp2_ftr_emss + sube3a.scp1_ftr_emss)*1000 AS avg_emissions
      FROM bu_esg_work.PCAF_fixed AS sube3a
      WHERE sube3a.nome_pais = 'Portugal'
        AND sube3a.classe_ativo = 'Commercial real estate'
        AND sube3a.unidade_fator_de_emissao = 'tCO2e/m2'
        AND sube3a.ano_fator_emissao = '2023'
      GROUP BY sube3a.epc,
               sube3a.Tipo_Finalidade_Bem_fixed) AS e3a ON e3a.epc = t.Certeficado_Energetico
   AND e3a.Tipo_Finalidade_Bem_fixed = t.Tipo_Imovel_Desc_PCAF
   LEFT JOIN
     (SELECT sube3.epc,
             sube3.Tipo_Finalidade_Bem_fixed,
             AVG(sube3.scp2_ftr_emss + sube3.scp1_ftr_emss)*1000 AS avg_emissions_building
      FROM bu_esg_work.PCAF_fixed AS sube3
      WHERE sube3.nome_pais = 'Portugal'
        AND sube3.classe_ativo = 'Commercial real estate'
        AND sube3.unidade_fator_de_emissao = 'tCO2e/Building'
        AND sube3.ano_fator_emissao = '2023'
      GROUP BY sube3.epc,
               sube3.Tipo_Finalidade_Bem_fixed) AS e3 ON e3.epc = t.Certeficado_Energetico
   AND e3.Tipo_Finalidade_Bem_fixed = t.Tipo_Imovel_Desc_PCAF
   LEFT JOIN
     (SELECT sube4.Tipo_Finalidade_Bem_fixed,
             AVG(sube4.scp2_ftr_emss + sube4.scp1_ftr_emss)*1000 AS avg_emissions
      FROM bu_esg_work.PCAF_fixed AS sube4
      WHERE sube4.nome_pais = 'Portugal'
        AND sube4.classe_ativo = 'Commercial real estate'
        AND sube4.unidade_fator_de_emissao = 'tCO2e/m2'
        AND sube4.ano_fator_emissao = '2023'
      GROUP BY sube4.Tipo_Finalidade_Bem_fixed) AS e4 ON e4.Tipo_Finalidade_Bem_fixed = t.Tipo_Imovel_Desc_PCAF
   LEFT JOIN
     (SELECT sube5.Tipo_Finalidade_Bem_fixed,
             AVG(sube5.scp2_ftr_emss + sube5.scp1_ftr_emss)*1000 AS avg_emissions_building
      FROM bu_esg_work.PCAF_fixed AS sube5
      WHERE sube5.nome_pais = 'Portugal'
        AND sube5.classe_ativo = 'Commercial real estate'
        AND sube5.unidade_fator_de_emissao = 'tCO2e/Building'
        AND sube5.ano_fator_emissao = '2023'
      GROUP BY sube5.Tipo_Finalidade_Bem_fixed) AS e5 ON e5.Tipo_Finalidade_Bem_fixed = t.Tipo_Imovel_Desc_PCAF) AS p
	  
	  LEFT JOIN (SELECT distinct load_id + 1 as max_load_id,
					    1 as dummy_id
				 FROM bu_esg_work.CRE_Emissions
				 where load_id = (select max(load_id) from bu_esg_work.CRE_Emissions)
				 ) as lid
	  ON p.dummy_id = lid.dummy_id;
