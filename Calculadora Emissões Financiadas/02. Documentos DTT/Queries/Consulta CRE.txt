
--Seleciona dados da ultima corrida

select * from bu_esg_work.CRE_Emissions where load_id = (select max(load_id) from bu_esg_work.CRE_Emissions);


--Calcula total de emissoes produzidas

select ref_date,description, load_id ,sum(financed_emissions) as Total_Financed_Emissions_CRE, sum(calculated_emissions_building) as Total_Building_Emissions_CRE from bu_esg_work.CRE_Emissions group by ref_date,description,load_id;
