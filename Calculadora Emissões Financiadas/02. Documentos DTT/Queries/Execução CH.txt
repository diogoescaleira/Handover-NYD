INSERT INTO TABLE bu_esg_work.CH_Emissions
SELECT p.ref_date,
	   from_unixtime(unix_timestamp()) as execution_date,
	   case when lid.max_load_id is null then 1
	   else lid.max_load_id
	   END as load_id,
	   '${descri}' as description,
	   p.cempcta,
	   p.ckbalcao,
	   p.zdeposit,
	   p.zcliente,
	   p.cknumcta,
	   p.ckbalbem,
	   p.ckctabem,
	   p.ckrefbem,
	   p.ddvencim,
	   p.cod_concelho,
	   p.area_util_pcaf as area_util,
	   p.ienerg_pcaf as ienerg,
	   p.score_enrg,
	   p.mavalbem as financed_amount,
	   p.gross_carrying_amount_bem as gross_carrying_amount,
	   p.atribution_factor,
	   p.calculated_emissions_m2,
	   p.Calculated_emissions_building,
	   p.Calculated_emissions_building * p.ATRIBUTION_FACTOR AS Financed_Emissions
FROM
  (SELECT t.*,
	  1 as dummy_id ,
		  CASE
			  WHEN t.SCORE_ENRG = '3a' THEN e3a.avg_emissions
			  WHEN t.SCORE_ENRG = '3' THEN NULL
			  WHEN t.SCORE_ENRG = '4' THEN e4.avg_emissions4
			  WHEN t.SCORE_ENRG = '5' THEN NULL
			  ELSE NULL
		  END AS Calculated_Emissions_m2,
		  CASE
			  WHEN t.SCORE_ENRG = '3a' THEN e3a.avg_emissions * t.area_util_pcaf
			  WHEN t.SCORE_ENRG = '3' THEN e3.avg_emissions_bulding
			  WHEN t.SCORE_ENRG = '4' THEN e4.avg_emissions4 * t.area_util_pcaf
			  WHEN t.SCORE_ENRG = '5' THEN e5.avg_emissions5
			  ELSE NULL
		  END AS Calculated_emissions_building
   FROM
	 (SELECT p.*,
			 CASE
				 WHEN p.gross_carrying_amount_bem > p.Origin_Amount THEN 1
				 ELSE p.gross_carrying_amount_bem / p.Origin_Amount
			 END AS atribution_factor,
			 CASE
				 WHEN (p.ienerg_PCAF IS NOT NULL
					   AND p.ienerg_PCAF != '0'
					   AND p.area_util_pcaf IS NOT NULL
					   AND p.area_util_pcaf != 0) THEN '3a'
				 WHEN (p.ienerg_PCAF IS NOT NULL
					   AND p.ienerg_PCAF != '0') THEN '3'
				 WHEN (p.area_util_pcaf IS NOT NULL
					   AND p.area_util_pcaf != 0) THEN '4'
				 WHEN (p.area_util_pcaf IS NULL
					   OR p.area_util_pcaf = 0) THEN '5'
				 ELSE 'ERROR'
			 END AS SCORE_ENRG,
			 'Residential real estate' AS classe
	  FROM
		(SELECT h.*,
				(h.Origin_Amount / SUM(h.Origin_Amount) OVER (PARTITION BY h.cempcta, h.ckbalcao, h.zdeposit, h.zcliente, h.cknumcta)) * h.sum_gross_carrying_amount AS gross_carrying_amount_bem,
				CASE
					WHEN h.ienerg IS NULL
						 OR h.ienerg = '' THEN h.clase_energetica
					WHEN h.ienerg = '+' THEN 'A+'
					WHEN h.ienerg = '-' THEN 'B-'
					WHEN h.ienerg = 'G' THEN 'F'
					WHEN h.ienerg = 'J'
						 OR h.ienerg = 'S'
						 OR h.ienerg = 'N'
						 OR h.ienerg IS NULL THEN NULL
					ELSE NULL
				END AS ienerg_PCAF,
				CASE
					WHEN h.area_util IS NOT NULL
						 OR h.area_util > 0 THEN h.area_util
					ELSE h.narutil
				END AS area_util_pcaf
		 FROM
		   (SELECT MAX(a.ref_date) AS ref_date,
				   MAX(a.cempcta) AS cempcta,
				   MAX(a.ckbalcao) AS ckbalcao,
				   MAX(a.zdeposit) AS zdeposit,
				   MAX(a.zcliente) AS zcliente,
				   MAX(a.cknumcta) AS cknumcta,
				   MAX(a.tipo_contrato) AS tipo_contrato,
				   MAX(d.ckbalbem) AS ckbalbem,
				   MAX(d.ckctabem) AS ckctabem,
				   MAX(d.ckrefbem) AS ckrefbem,
				   MAX(d.cod_finalidade) AS cod_finalidade,
				   MAX(a.saldo) AS sum_gross_carrying_amount,
				   MAX(a.ddvencim) AS ddvencim,
				   REPLACE(MAX(d.ienerg), ' ', '') AS ienerg,
				   MAX(d.cdisconf) AS cdisconf,
				   MAX(d.cod_concelho) AS cod_concelho,
				   MAX(d.mavalia) AS Origin_Amount,
				   MAX(d.narutil) AS area_util,
				   MAX(g.`32_type_collateral`) AS `32_type_collateral`,
				   MAX(g.consumos) AS consumos,
				   MAX(g.clase_energetica) AS clase_energetica,
				   MAX(g.emisiones) AS emisiones,
				   MAX(g.af) AS af,
				   MAX(g.narutil) AS narutil,
				   MAX(g.`13_gross_carrying_amount`) AS `13_gross_carrying_amount`,
				   MAX(g.mavalbem) AS mavalbem
			FROM
			  (SELECT a1.ref_date,
					  a1.cempcta,
					  a1.ckbalcao,
					  a1.zdeposit,
					  a1.zcliente,
					  a1.cknumcta,
					  a1.tipo_contrato,
					  SUM(a1.c007) AS saldo,
					  a2.ddvencim
			   FROM cd_captools.ct209_rent_kpm AS a1
			   LEFT JOIN cd_captools.ct004_univ_cto AS a2 ON a1.ref_date = a2.ref_date
			   AND a1.cempcta = a2.cempresa
			   AND a1.ckbalcao = a2.cbalcao
			   AND a1.zdeposit = a2.zdeposit
			   AND a1.zcliente = a2.zcliente
			   AND a1.cknumcta = a2.cnumecta
			   WHERE a1.ref_date = '${ref_date}'
				 AND SUBSTRING(a1.ckmetamis,1,4) = 'AI03'
				 AND a1.c007 <> 0
			   GROUP BY a1.ref_date,
						a1.cempcta,
						a1.ckbalcao,
						a1.zdeposit,
						a1.zcliente,
						a1.cknumcta,
						a1.tipo_contrato,
						a2.ddvencim) AS a
			LEFT JOIN cd_emprestimos.gpt31_contrato AS c ON a.ckbalcao = c.ckbalcao
			AND a.cknumcta = c.cknumcta
			LEFT JOIN
			  (SELECT DISTINCT d1.cempresp,
							   d1.cempresa,
							   d1.ckbalres,
							   d1.ckctares,
							   d1.ckbalbem,
							   d1.ckctabem,
							   d1.ckrefbem,
							   d1.mavalia,
							   d1.montante,
							   d1.cfinbem,
							   d7.tayd91c0_gelemtab AS cod_finalidade,
							   d2.cdisconf,
							   d6.tayd91c0_gelemtab AS cod_concelho,
							   d3.narutil,
							   d5.ienerg
			   FROM cd_garantias.gt018_rateio_aval AS d1
			   LEFT JOIN cd_garantias.gt009_bens_imov AS d2 ON d1.cempbem=d2.cempbem
			   AND d1.ckbalbem=d2.ckbalbem
			   AND d1.ckctabem=d2.ckctabem
			   AND d1.ckrefbem = d2.ckrefbem
			   AND d1.ref_date = d2.ref_date
			   LEFT JOIN cd_emprestimos.gpt32_bemimove AS d3 ON d1.ckbalbem = d3.ckbalbem
			   AND d1.ckctabem= d3.ckctabem
			   LEFT JOIN cd_emprestimos.gpt31_contrato AS d5 ON d1.ckbalcao = d5.ckbalcao
			   AND d1.cknumcta = d5.cknumcta
			   LEFT JOIN
				 (SELECT j.*
				  FROM cd_estruturais.tat91_tabelas AS j
				  WHERE j.data_date_part =
					  ( SELECT data_date_part
					   FROM cd_estruturais.tat91_tabelas
					   WHERE DATEDIFF('${ref_date}', data_date_part) < 0
					   ORDER BY DATEDIFF('${ref_date}', data_date_part) ASC
					   LIMIT 1)
					AND j.tayd91c0_ctabela = 'J48') AS d6 ON d6.tayd91c0_celemtab = CONCAT(SUBSTRING(d2.cdisconf,1,4),'00')
			   LEFT JOIN
				 (SELECT r.*
				  FROM cd_estruturais.tat91_tabelas AS r
				  WHERE r.data_date_part =
					  ( SELECT data_date_part
					   FROM cd_estruturais.tat91_tabelas
					   WHERE DATEDIFF('${ref_date}', data_date_part) < 0
					   ORDER BY DATEDIFF('${ref_date}', data_date_part) ASC
					   LIMIT 1)
					AND r.tayd91c0_ctabela = '438') AS d7 ON d7.tayd91c0_celemtab = d1.cfinbem
			   WHERE d1.ref_date='${ref_date}'
				 AND d1.tipologia = 'IMO') AS d ON d.ckbalres = a.ckbalcao
			AND d.cempresp = a.cempcta
			AND d.ckctares = a.cknumcta
			LEFT JOIN
			  (SELECT DISTINCT g1.cempresa_ct,
							   g1.cbalcao_ct,
							   g1.cnumecta_ct,
							   g1.zdeposit_ct,
							   g1.ckbalbem,
							   g1.ckctabem,
							   g1.ckrefbem,
							   g1.`32_type_collateral`,
							   g1.consumos,
							   g1.clase_energetica,
							   g1.emisiones,
							   g1.af,
							   g1.narutil,
							   g2.`13_gross_carrying_amount`,
							   g2.mavalbem
			   FROM bu_esg_work.contrto_gloval_clase_energetica_dez23 AS g1
			   LEFT JOIN bu_esg_work.reparticao_garantias_final_dez23 AS g2 ON g1.cempresa_ct = g2.cempresa_ct
			   AND g1.cbalcao_ct = g2.cbalcao_ct
			   AND g1.cnumecta_ct= g2.cnumecta_ct
			   AND g1.zdeposit_ct=g2.zdeposit_ct
			   AND g1.ckbalbem= g2.ckbalbem
			   AND g1.ckctabem = g2.ckctabem
			   AND g1.ckrefbem = g2.ckrefbem
			   WHERE g1.`32_type_collateral` IS NOT NULL) AS g ON g.cempresa_ct = a.cempcta
			AND g.cbalcao_ct = a.ckbalcao
			AND g.cnumecta_ct= a.cknumcta
			AND g.zdeposit_ct=a.zdeposit
			AND g.ckbalbem= d.ckbalbem
			AND g.ckctabem = d.ckctabem
			AND g.ckrefbem = d.ckrefbem
			LEFT JOIN
			  (SELECT gpt.csitbemic,
					  gpt.ckbalbem,
					  gpt.ckctabem
			   FROM cd_emprestimos.gpt18_bens AS gpt
			   WHERE gpt.data_date_part =
				   (SELECT data_date_part
					FROM cd_emprestimos.gpt18_bens
					WHERE DATEDIFF('${ref_date}', data_date_part) > 0
					ORDER BY DATEDIFF('${ref_date}', data_date_part) ASC
					LIMIT 1)) AS p ON p.ckbalbem = D.ckbalbem
			AND p.ckctabem = D.ckctabem
			WHERE p.csitbemic IN ('01',
								  '03',
								  '04')
			GROUP BY a.cempcta,
					 A.CKBALCAO,
					 a.zdeposit,
					 a.cknumcta,
					 d.ckbalbem,
					 d.ckctabem,
					 d.ckrefbem) AS h) AS p) AS t
   LEFT JOIN
	 (SELECT sube3a.epc,
			 AVG(sube3a.scp2_ftr_emss + sube3a.scp1_ftr_emss)*1000 AS avg_emissions
	  FROM bu_esg_work.PCAF_fixed AS sube3a
	  WHERE sube3a.nome_pais = 'Portugal'
		AND sube3a.classe_ativo = 'Residential real estate'
		AND sube3a.unidade_fator_de_emissao = 'tCO2e/m2'
		AND sube3a.ano_fator_emissao = '2023'
	  GROUP BY sube3a.epc) AS e3a ON e3a.epc = t.ienerg_pcaf
   LEFT JOIN
	 (SELECT sube3.epc AS certi,
			 AVG(sube3.scp2_ftr_emss + sube3.scp1_ftr_emss)*1000 AS avg_emissions_bulding
	  FROM bu_esg_work.PCAF_fixed AS sube3
	  WHERE sube3.nome_pais = 'Portugal'
		AND sube3.classe_ativo = 'Residential real estate'
		AND sube3.unidade_fator_de_emissao = 'tCO2e/Building'
		AND sube3.ano_fator_emissao = '2023'
	  GROUP BY sube3.epc) AS e3 ON e3.certi = t.ienerg_pcaf
   LEFT JOIN
	 (SELECT sube4.classe_ativo,
			 AVG(sube4.scp2_ftr_emss + sube4.scp1_ftr_emss)*1000 AS avg_emissions4
	  FROM bu_esg_work.PCAF_fixed AS sube4
	  WHERE sube4.nome_pais = 'Portugal'
		AND sube4.classe_ativo = 'Residential real estate'
		AND sube4.unidade_fator_de_emissao = 'tCO2e/m2'
		AND sube4.ano_fator_emissao = '2023'
	  GROUP BY sube4.classe_ativo) AS e4 ON e4.classe_ativo = t.classe
   LEFT JOIN
	 (SELECT sube5.classe_ativo AS classe_ativo5,
			 AVG(sube5.scp2_ftr_emss + sube5.scp1_ftr_emss)*1000 AS avg_emissions5
	  FROM bu_esg_work.PCAF_fixed AS sube5
	  WHERE sube5.nome_pais = 'Portugal'
		AND sube5.classe_ativo = 'Residential real estate'
		AND sube5.unidade_fator_de_emissao = 'tCO2e/Building'
		AND sube5.ano_fator_emissao = '2023'
	  GROUP BY sube5.classe_ativo) AS e5 ON e5.classe_ativo5 = t.classe) AS p
	  
	  LEFT JOIN (SELECT distinct load_id + 1 as max_load_id,
					    1 as dummy_id
				 FROM bu_esg_work.CH_Emissions
				 where load_id = (select max(load_id) from bu_esg_work.CH_Emissions)
				 ) as lid
	  ON p.dummy_id = lid.dummy_id;
