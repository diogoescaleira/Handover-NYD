-- VALIDAÇÃO UNIVERSO E CAMPOS: SIMULAÇÃO VS EXECUÇÃO IT

WITH AUX AS (

SELECT
    CASE
        WHEN SIMUL.PRTCTN_ID IS NULL THEN 'EXCLUSIVO IT'
        WHEN IT.PRTCTN_ID IS NULL THEN 'EXCLUSIVO SIMUL'
        ELSE 'COMUM'
    END AS UNIVERSO,
    
    COALESCE(SIMUL.COD_SOC, IT.COD_SOC) AS COD_SOC,
    COALESCE(SIMUL.PRTCTN_ID, IT.PRTCTN_ID) AS PRTCTN_ID,
    
    SIMUL.TYP_PRTCTN AS TYP_PRTCTN_SIMUL,
    IT.TYP_PRTCTN AS TYP_PRTCTN_IT,
    CASE WHEN SIMUL.TYP_PRTCTN <=> IT.TYP_PRTCTN THEN 'OK' ELSE 'NOK' END AS VAL_TYP_PRTCTN,
    
    SIMUL.PRTCTN_VL AS PRTCTN_VL_SIMUL,
    IT.PRTCTN_VL AS PRTCTN_VL_IT,
    CASE WHEN SIMUL.PRTCTN_VL <=> IT.PRTCTN_VL THEN 'OK' ELSE 'NOK' END AS VAL_PRTCTN_VL,
    
    SIMUL.TYP_VHCL AS TYP_VHCL_SIMUL,
    IT.TYP_VHCL AS TYP_VHCL_IT,
    CASE WHEN SIMUL.TYP_VHCL <=> IT.TYP_VHCL THEN 'OK' ELSE 'NOK' END AS VAL_TYP_VHCL,
    
    SIMUL.FLG_ELCTRC_VHCL AS FLG_ELCTRC_VHCL_SIMUL,
    IT.FLG_ELCTRC_VHCL AS FLG_ELCTRC_VHCL_IT,
    CASE WHEN SIMUL.FLG_ELCTRC_VHCL <=> IT.FLG_ELCTRC_VHCL THEN 'OK' ELSE 'NOK' END AS VAL_FLG_ELCTRC_VHCL
    
    
FROM BU_CAPTOOLS_WORK.LT004_GD_MAR25_DG SIMUL
FULL JOIN
    (
    SELECT COD_SOC, REF_DATE, PRTCTN_ID, TYP_PRTCTN, PRTCTN_VL, TYP_VHCL, FLG_ELCTRC_VHCL
    FROM BU_LOANTAPE_WORK.LT004_PROTECTION
    WHERE REF_DATE = '${DATE}'
    AND AMBITO = 'GD'
    AND SEGMENTO = 'ESTR'
    AND NOME_PERIMETRO = 'Individual IDComb'
    ) IT

ON SIMUL.PRTCTN_ID=IT.PRTCTN_ID
)


SELECT UNIVERSO, VAL_TYP_VHCL, COUNT(*)

-- VALIDAÇÃO UNIVERSO E CAMPOS: SIMULAÇÃO VS EXECUÇÃO IT

WITH AUX AS (

SELECT
    CASE
        WHEN GD.PRTCTN_ID IS NULL THEN 'EXCLUSIVO PILAR3'
        WHEN PILAR3.PRTCTN_ID IS NULL THEN 'EXCLUSIVO GD'
        ELSE 'COMUM'
    END AS UNIVERSO,
    
    COALESCE(GD.COD_SOC, PILAR3.COD_SOC) AS COD_SOC,
    COALESCE(GD.PRTCTN_ID, PILAR3.PRTCTN_ID) AS PRTCTN_ID,
    
    GD.TYP_PRTCTN AS TYP_PRTCTN_GD,
    PILAR3.TYP_PRTCTN AS TYP_PRTCTN_PILAR3,
    CASE WHEN GD.TYP_PRTCTN <=> PILAR3.TYP_PRTCTN THEN 'OK' ELSE 'NOK' END AS VAL_TYP_PRTCTN,
    
    GD.PRTCTN_VL AS PRTCTN_VL_GD,
    PILAR3.PRTCTN_VL AS PRTCTN_VL_PILAR3,
    CASE WHEN GD.PRTCTN_VL <=> PILAR3.PRTCTN_VL THEN 'OK' ELSE 'NOK' END AS VAL_PRTCTN_VL,
    
    GD.TYP_VHCL AS TYP_VHCL_GD,
    PILAR3.TYP_VHCL AS TYP_VHCL_PILAR3,
    CASE WHEN GD.TYP_VHCL <=> PILAR3.TYP_VHCL THEN 'OK' ELSE 'NOK' END AS VAL_TYP_VHCL,
    
    GD.FLG_ELCTRC_VHCL AS FLG_ELCTRC_VHCL_GD,
    PILAR3.FLG_ELCTRC_VHCL AS FLG_ELCTRC_VHCL_PILAR3,
    CASE WHEN GD.FLG_ELCTRC_VHCL <=> PILAR3.FLG_ELCTRC_VHCL THEN 'OK' ELSE 'NOK' END AS VAL_FLG_ELCTRC_VHCL
    
    
FROM 
    (
    SELECT COD_SOC, REF_DATE, PRTCTN_ID, TYP_PRTCTN, PRTCTN_VL, TYP_VHCL, FLG_ELCTRC_VHCL
    FROM BU_LOANTAPE_WORK.LT004_PROTECTION
    WHERE REF_DATE = '${DATE}'
    AND AMBITO = 'GD'
    AND SEGMENTO = 'ESTR'
    AND NOME_PERIMETRO = 'Individual Corporativo'
    ) GD
    
FULL JOIN
    (
    SELECT COD_SOC, REF_DATE, PRTCTN_ID, TYP_PRTCTN, PRTCTN_VL, TYP_VHCL, FLG_ELCTRC_VHCL
    FROM BU_LOANTAPE_WORK.LT004_PROTECTION
    WHERE REF_DATE = '${DATE}'
    AND AMBITO = 'PILAR3'
    AND SEGMENTO = 'ESTR'
    AND NOME_PERIMETRO = 'Individual Corporativo'
    ) PILAR3

ON GD.PRTCTN_ID=PILAR3.PRTCTN_ID
)


SELECT UNIVERSO, VAL_PRTCTN_VL, COUNT(*)

-- PRTCTN_ID, TYP_VHCL_IT, FLG_ELCTRC_VHCL_IT, FLG_ELCTRC_VHCL_SIMUL
