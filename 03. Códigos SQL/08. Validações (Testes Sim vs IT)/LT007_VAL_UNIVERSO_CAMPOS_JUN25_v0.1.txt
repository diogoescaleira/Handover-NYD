-- AMBITO GD VS AMBITO PILAR 3 (EXECUÇÃO IT)

WITH AUX AS (
    SELECT
        CASE
            WHEN GD.INSTRMNT_ID IS NULL THEN 'EXCLUSIVO PILAR3'
            WHEN PILAR.INSTRMNT_ID IS NULL THEN 'EXCLUSIVO GD'
            ELSE 'COMUM'
        END AS UNIVERSO,
        COALESCE(GD.REF_DATE, PILAR.REF_DATE) AS REF_DATE,
        COALESCE(GD.COD_SOC, PILAR.COD_SOC) AS COD_SOC,
        COALESCE(GD.INSTRMNT_ID, PILAR.INSTRMNT_ID) AS INSTRMNT_ID,
        COALESCE(GD.SSTNBL_INDCTR, PILAR.SSTNBL_INDCTR) AS SSTNBL_INDCTR,
        
        GD.SSTNBL_CTGRY AS SSTNBL_CTGRY_GD,
        PILAR.SSTNBL_CTGRY AS SSTNBL_CTGRY_PILAR,
        CASE WHEN GD.SSTNBL_CTGRY <=> PILAR.SSTNBL_CTGRY THEN 'OK' ELSE 'NOK' END AS VAL_SSTNBL_CTGRY,
        
        GD.PRCNTG_ACTVTY AS PRCNTG_ACTVTY_GD,
        PILAR.PRCNTG_ACTVTY AS PRCNTG_ACTVTY_PILAR,
        CASE WHEN GD.PRCNTG_ACTVTY <=> PILAR.PRCNTG_ACTVTY THEN 'OK' ELSE 'NOK' END AS VAL_PRCNTG_ACTVTY  
        
    
    FROM 
        (
        SELECT DISTINCT REF_DATE, COD_SOC, INSTRMNT_ID, SSTNBL_INDCTR, SSTNBL_CTGRY, PRCNTG_ACTVTY
        FROM BU_LOANTAPE_WORK.LT007_INSTR_ACTIVITY
        WHERE REF_DATE = '${DATE}'
        AND AMBITO = 'GD'
        AND SEGMENTO = 'ESTR'
        AND NOME_PERIMETRO = 'Individual IDComb'
        ) GD
    
    FULL JOIN
    
        (
        SELECT DISTINCT REF_DATE, COD_SOC, INSTRMNT_ID, SSTNBL_INDCTR, SSTNBL_CTGRY, PRCNTG_ACTVTY
        FROM BU_LOANTAPE_WORK.LT007_INSTR_ACTIVITY
        WHERE REF_DATE = '${DATE}'
        AND AMBITO = 'PILAR3'
        AND SEGMENTO = 'ESTR'
        AND NOME_PERIMETRO = 'Individual IDComb'
        ) PILAR
    
    ON GD.INSTRMNT_ID=PILAR.INSTRMNT_ID
    AND GD.SSTNBL_INDCTR=PILAR.SSTNBL_INDCTR
) 

SELECT 
    UNIVERSO,
    INSTRMNT_ID,
    SSTNBL_INDCTR
    -- SSTNBL_INDCTR,
    -- SSTNBL_CTGRY_GD,
    -- SSTNBL_CTGRY_PILAR,
    -- VAL_SSTNBL_CTGRY
    -- COUNT(*) AS COUNT_REG
-- VAL_SSTNBL_CTGRY, VAL_PRCNTG_ACTVTY, COUNT(*) AS COUNT_REG
FROM AUX
WHERE UNIVERSO = 'EXCLUSIVO PILAR3'
-- AND VAL_SSTNBL_CTGRY = 'NOK'
-- GROUP BY 1
LIMIT 10
;
