;

-- EXECUÇÃO IT
-- # REGISTOS = 800.334 (CONSIDERANDO APENAS OS CAMPOS NECESSÁRIOS PARA GD)

SELECT DISTINCT COD_SOC, REF_DATE, INSTRMNT_ID, PRTCTN_ID, TYP_RLTN
FROM BU_LOANTAPE_WORK.LT006_INSTR_PROT
WHERE REF_DATE = '${DATE}'
AND AMBITO = 'GD'
AND SEGMENTO = 'ESTR'
AND NOME_PERIMETRO = 'Individual Corporativo'
;


-- VALIDAÇÃO UNIVERSO E CAMPOS

WITH AUX AS (

SELECT
    CASE
        WHEN SIMUL.PRTCTN_ID IS NULL THEN 'EXCLUSIVO IT'
        WHEN IT.PRTCTN_ID IS NULL THEN 'EXCLUSIVO SIMUL'
        ELSE 'COMUM'
    END AS UNIVERSO,
    
    COALESCE(SIMUL.COD_SOC, IT.COD_SOC) AS COD_SOC,
    COALESCE(SIMUL.INSTRMNT_ID, IT.INSTRMNT_ID) AS INSTRMNT_ID,
    COALESCE(SIMUL.PRTCTN_ID, IT.PRTCTN_ID) AS PRTCTN_ID,
    
    SIMUL.TYP_RLTN AS TYP_RLTN_SIMUL,
    IT.TYP_RLTN AS TYP_RLTN_IT,
    CASE WHEN SIMUL.TYP_RLTN <=> IT.TYP_RLTN THEN 'OK' ELSE 'NOK' END AS VAL_TYP_RLTN
  
    
FROM BU_CAPTOOLS_WORK.LT006_GD_JUN25_FC SIMUL

FULL JOIN
    (
    SELECT DISTINCT COD_SOC, REF_DATE, INSTRMNT_ID, PRTCTN_ID, TYP_RLTN
    FROM BU_LOANTAPE_WORK.LT006_INSTR_PROT
    WHERE REF_DATE = '${DATE}'
    AND AMBITO = 'GD'
    AND SEGMENTO = 'ESTR'
    AND NOME_PERIMETRO = 'Individual IDComb'
    ) IT

ON SIMUL.PRTCTN_ID=IT.PRTCTN_ID
AND SIMUL.INSTRMNT_ID=IT.INSTRMNT_ID
)

SELECT * 
-- UNIVERSO, TYP_RLTN_SIMUL, TYP_RLTN_IT, VAL_TYP_RLTN, COUNT(*)
FROM AUX
WHERE VAL_TYP_RLTN = 'NOK'
-- GROUP BY 1,2,3,4
-- WHERE INSTRMNT_ID = '82000000001018111000430000739764'
;

-- AMBITO GD VS AMBITO PILAR 3 (EXECUÇÃO IT)

WITH AUX AS (
SELECT
    CASE
        WHEN GD.PRTCTN_ID IS NULL THEN 'EXCLUSIVO PILAR3'
        WHEN PILAR.PRTCTN_ID IS NULL THEN 'EXCLUSIVO GD'
        ELSE 'COMUM'
    END AS UNIVERSO,
    
    COALESCE(GD.COD_SOC, PILAR.COD_SOC) AS COD_SOC,
    COALESCE(GD.INSTRMNT_ID, PILAR.INSTRMNT_ID) AS INSTRMNT_ID,
    COALESCE(GD.PRTCTN_ID, PILAR.PRTCTN_ID) AS PRTCTN_ID,
    
    GD.TYP_RLTN AS TYP_RLTN_GD,
    PILAR.TYP_RLTN AS TYP_RLTN_PILAR,
    CASE WHEN GD.TYP_RLTN <=> PILAR.TYP_RLTN THEN 'OK' ELSE 'NOK' END AS VAL_TYP_RLTN

FROM 
    (
    SELECT DISTINCT COD_SOC, REF_DATE, INSTRMNT_ID, PRTCTN_ID, TYP_RLTN
    FROM BU_LOANTAPE_WORK.LT006_INSTR_PROT
    WHERE REF_DATE = '${DATE}'
    AND AMBITO = 'GD'
    AND SEGMENTO = 'ESTR'
    AND NOME_PERIMETRO = 'Individual IDComb'
    ) GD

FULL JOIN

    (
    SELECT DISTINCT COD_SOC, REF_DATE, INSTRMNT_ID, PRTCTN_ID, TYP_RLTN
    FROM BU_LOANTAPE_WORK.LT006_INSTR_PROT
    WHERE REF_DATE = '${DATE}'
    AND AMBITO = 'PILAR3'
    AND SEGMENTO = 'ESTR'
    AND NOME_PERIMETRO = 'Individual IDComb'
    ) PILAR

ON GD.INSTRMNT_ID=PILAR.INSTRMNT_ID
AND GD.PRTCTN_ID=PILAR.PRTCTN_ID
) 

SELECT UNIVERSO, VAL_TYP_RLTN, COUNT(*)
FROM AUX
GROUP BY 1,2
;
