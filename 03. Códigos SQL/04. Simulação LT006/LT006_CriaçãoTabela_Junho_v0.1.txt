DROP TABLE IF EXISTS BU_CAPTOOLS_WORK.LT006_GD_JUN25_FC_AUX1;

CREATE TABLE BU_CAPTOOLS_WORK.LT006_GD_JUN25_FC_AUX1 AS
-- WITH AA AS (
SELECT 
    ORIGEM,
    COD_SOC,
    REF_DATE AS DT_RFRNC,
    INSTRMNT_ID,
    CHAVE_BEM,
    PRTCTN_ID
FROM

    (
    SELECT 
        INSTRMNT_ID,
        CHAVE_BEM,
        CHAVE_BEM_PRE as PRTCTN_ID,
        ORIGEM,
        ROW_NUMBER() OVER (PARTITION BY CONCAT(INSTRMNT_ID,CHAVE_BEM) ORDER BY CASE WHEN ORIGEM='MUG' OR ORIGEM='PM-F' OR ORIGEM='PM-I' OR ORIGEM='PM-P' THEN 1 ELSE 0 END DESC) AS ROW_1,
        COD_SOC,
        REF_DATE
    FROM
    
        (
        SELECT DISTINCT
            LT002.INSTRMNT_ID,
            GT012.CHAVE_BEM,
            GT012.CHAVE_BEM_PRE,'MUG' AS ORIGEM,
            COD_SOC,
            REF_DATE
        FROM
            (
            SELECT INSTRMNT_ID, COD_SOC, REF_DATE
            FROM BU_LOANTAPE_WORK.LT002_INSTRUMENT
            WHERE REF_DATE='${REF_DATE}'
            AND NOME_PERIMETRO= 'Individual IDComb'
            AND SEGMENTO = 'ESTR' 
            AND AMBITO = 'PM'
            AND ENTIDADE IN ('00100', '00280') -- [17/JUN]: ADICIONADA A ENTIDADE 00280 POI GD TBM REPORTA TOTTA IRELAND
            ) LT002
        
        INNER JOIN
            (
            SELECT DISTINCT 
                CONCAT (CEMPRESA_FR, CBALCAO_FR, CNUMECTA_FR, ZDEPOSIT_FR) AS CHAVE_FINREP,
                CONCAT (CEMPRESA, CBALCAO, CNUMECTA, ZDEPOSIT) AS CHAVE_MASTER
            FROM CD_CAPTOOLS.KT_CHAVES_FINREP
            WHERE REF_DATE='${REF_DATE}'
            ) FR 
        
        ON LT002.INSTRMNT_ID = FR.CHAVE_FINREP
        
        INNER JOIN 
            (
            SELECT DISTINCT 
                CONCAT (CEMPRESP, CKBALRES, CKCTARES, CKREFRESP) AS CHAVE_MASTER,
                CONCAT (CEMPRESA, CKBALCAO, CKNUMCTA, COMP_ID_GAR) AS CHAVE_PROCESSO
            FROM CD_GARANTIAS.GT001_TRZ_RSP_GT
            WHERE REF_DATE='${REF_DATE}'
            ) GT001 
        
        ON FR.CHAVE_MASTER = GT001.CHAVE_MASTER
            
        INNER JOIN
            (
            SELECT DISTINCT 
                CONCAT (CEMPRESA, CKBALCAO, CKNUMCTA, COMP_ID_GAR) AS CHAVE_PROCESSO,
                CONCAT (CEMPBEM, CKBALBEM, CKCTABEM, CKREFBEM) AS CHAVE_BEM,
                CONCAT ('M','_',CKBALBEM,'_',CKCTABEM,'_',CKREFBEM) AS CHAVE_BEM_PRE
            FROM CD_GARANTIAS.GT012_TRZ_GT_MOB
            WHERE REF_DATE='${REF_DATE}'
            ) GT012
        
        ON GT001.CHAVE_PROCESSO = GT012.CHAVE_PROCESSO
        
    UNION ALL
        
        SELECT DISTINCT LT002.INSTRMNT_ID, PROCESSO_BI.CHAVE_BEM, PROCESSO_BI.CHAVE_BEM_PRE,'MBI' AS ORIGEM, COD_SOC, REF_DATE
        FROM 
            (
            SELECT INSTRMNT_ID, COD_SOC, REF_DATE
            FROM BU_LOANTAPE_WORK.LT002_INSTRUMENT
            WHERE REF_DATE = '${REF_DATE}'
            AND NOME_PERIMETRO = 'Individual IDComb'
            AND SEGMENTO = 'ESTR' 
            AND AMBITO = 'PM'
            AND ENTIDADE IN ('00100', '00280') -- [17/JUN]: ADICIONADA A ENTIDADE 00280 POI GD TBM REPORTA TOTTA IRELAND
            ) LT002
            
        INNER JOIN 
            (
            SELECT DISTINCT 
                CONCAT (CEMPRESA_FR, CBALCAO_FR, CNUMECTA_FR, ZDEPOSIT_FR) AS CHAVE_FINREP,
                CONCAT (CEMPRESA, CBALCAO, CNUMECTA, ZDEPOSIT) AS CHAVE_MASTER
            FROM CD_CAPTOOLS.KT_CHAVES_FINREP
            WHERE REF_DATE='${REF_DATE}'
            ) FR
            
        ON LT002.INSTRMNT_ID = FR.CHAVE_FINREP
                
        INNER JOIN
            (
            SELECT DISTINCT 
                CONCAT (BANK_ID, BRANCH_CODE, CONTRACT_ID, REFERENCE_CODE) AS CHAVE_RESPONSABILIDADE,
                CONCAT (PROCESS_BANK_ID, PROCESS_BRANCH_CODE, PROCESS_ACCOUNT_ID, PROCESS_REFERENCE_CODE) AS CHAVE_PROCESSO
                
            FROM BU_ESG_WORK.RESPONSIBILITY_PROCESS_REL
            WHERE DATA_DATE_PART IN (SELECT MAX (DATA_DATE_PART) FROM BU_ESG_WORK.RESPONSIBILITY_PROCESS_REL  WHERE DATA_DATE_PART <= '${REF_DATE}') 
            ) CREDITO_PROCESSO
            
        ON CREDITO_PROCESSO.CHAVE_RESPONSABILIDADE = FR.CHAVE_MASTER
        
        INNER JOIN
            (
            SELECT DISTINCT 
                CONCAT (PROCESS_BANK_ID, PROCESS_BRANCH_CODE, PROCESS_ACCOUNT_ID, PROCESS_REFERENCE_CODE) AS CHAVE_PROCESSO,
                CONCAT ('I','_',PROPERTY_BRANCH_CODE,'_',PROPERTY_CONTRACT_ID,'_',PROPERTY_REFERENCE_CODE) AS CHAVE_BEM_PRE,
                CONCAT (PROPERTY_BRANCH_CODE,PROPERTY_CONTRACT_ID,PROPERTY_REFERENCE_CODE) AS CHAVE_BEM,
                COLLATERAL_PROPERTY_IND AS FLG_COL,
                FINANCED_PROPERTY_IND AS FLG_FIN
                
            FROM BU_ESG_WORK.PROCESS_PROPERTY_REL
            WHERE DATA_DATE_PART IN (SELECT MAX (DATA_DATE_PART) FROM BU_ESG_WORK.PROCESS_PROPERTY_REL WHERE DATA_DATE_PART <= '${REF_DATE}') 
            AND (COLLATERAL_PROPERTY_IND = 1 OR FINANCED_PROPERTY_IND = 1)
            ) PROCESSO_BI
            
        ON CREDITO_PROCESSO.CHAVE_PROCESSO = PROCESSO_BI.CHAVE_PROCESSO
        
        UNION ALL

        SELECT DISTINCT 
            INSTRMNT_ID,
            REPLACE(SUBSTR(LT006.PRTCTN_ID,3,LENGTH(LT006.PRTCTN_ID)),'_','') AS CHAVE_BEM,
            PRTCTN_ID AS CHAVE_BEM_PRE,
            CONCAT('PM-',SUBSTR(LT006.PRTCTN_ID,1,1)) AS ORIGEM,
            COD_SOC,
            REF_DATE
            
        FROM
            (
            SELECT INSTRMNT_ID, PRTCTN_ID, COD_SOC, REF_DATE
            FROM BU_LOANTAPE_WORK.LT006_INSTR_PROT
            WHERE REF_DATE ='${REF_DATE}'
            AND NOME_PERIMETRO = 'Individual IDComb'
            AND SEGMENTO = 'ESTR' 
            AND AMBITO = 'PM'
            AND ENTIDADE = '00100'
            ) LT006
            
        )UNIV_TOTAL
        
    ) UNIV_TRAT
    
WHERE ROW_1 = 1
-- )

DROP TABLE IF EXISTS BU_CAPTOOLS_WORK.LT006_GD_JUN25_FC;

CREATE TABLE BU_CAPTOOLS_WORK.LT006_GD_JUN25_FC AS

SELECT DISTINCT
    DT_RFRNC,
    COD_SOC,
    INSTRMNT_ID,
    PRTCTN_ID,
    CASE
        WHEN SUBSTRING(PRTCTN_ID,1,1) = 'I' AND MAX_COLLATERAL_PROPERTY_IND = 1 AND MAX_FINANCED_PROPERTY_IND = 1 THEN 'REL01'
        WHEN SUBSTRING(PRTCTN_ID,1,1) = 'I' AND MAX_COLLATERAL_PROPERTY_IND = 1 AND MAX_FINANCED_PROPERTY_IND = 0 THEN 'REL02'
        WHEN SUBSTRING(PRTCTN_ID,1,1) = 'I' AND MAX_COLLATERAL_PROPERTY_IND = 0 AND MAX_FINANCED_PROPERTY_IND = 1 THEN 'REL03'
        WHEN SUBSTRING(PRTCTN_ID,1,1) = 'M' THEN 'REL01'
        ELSE 'REL02'
    END AS TYP_RLTN
        
FROM
    (
    SELECT 
        LT006.*,
        COLLATERAL_PROPERTY_IND,
        FINANCED_PROPERTY_IND,
        MAX(COLLATERAL_PROPERTY_IND) OVER (PARTITION BY LT006.INSTRMNT_ID, LT006.PRTCTN_ID) AS MAX_COLLATERAL_PROPERTY_IND,
        MAX(FINANCED_PROPERTY_IND) OVER (PARTITION BY LT006.INSTRMNT_ID, LT006.PRTCTN_ID) AS MAX_FINANCED_PROPERTY_IND
        
    FROM BU_CAPTOOLS_WORK.LT006_GD_JUN25_FC_AUX1 LT006
    
    LEFT JOIN
        (
        SELECT *
        FROM CD_CAPTOOLS.KT_CHAVES_FINREP
        WHERE REF_DATE = '${REF_DATE}'
        -- AND CONCAT(cempresa_fr, cbalcao_fr, cnumecta_FR, zdeposit_fr) = '8200000000696144200Y430000471836'
        ) FR
    
    ON SUBSTRING(LT006.INSTRMNT_ID,1,2) = FR.CEMPRESA_FR
    AND SUBSTRING(LT006.INSTRMNT_ID,3,4) = FR.CBALCAO_FR
    AND SUBSTRING(LT006.INSTRMNT_ID,7,11) = FR.CNUMECTA_FR
    AND SUBSTRING(LT006.INSTRMNT_ID,18,15) = FR.ZDEPOSIT_FR
    
    LEFT JOIN
        (
        SELECT *
        FROM BU_ESG_WORK.RESPONSIBILITY_PROCESS_REL
        WHERE DATA_DATE_PART IN (SELECT MAX (DATA_DATE_PART) FROM BU_ESG_WORK.RESPONSIBILITY_PROCESS_REL WHERE DATA_DATE_PART <= '${REF_DATE}')
        -- AND CONCAT(PROCESS_BRANCH_CODE,PROCESS_ACCOUNT_ID,PROCESS_REFERENCE_CODE) = '000000000000000000000069620522'
        -- AND CONCAT(BANK_ID,BRANCH_CODE,CONTRACT_ID,REFERENCE_CODE) = '8900000000696144200Y430000471836'
        ) RPR
    
    ON FR.CEMPRESA = RPR.BANK_ID
    AND FR.CBALCAO = RPR.BRANCH_CODE
    AND FR.CNUMECTA = RPR.CONTRACT_ID
    AND FR.ZDEPOSIT = RPR.REFERENCE_CODE
    
    LEFT JOIN
        (
        SELECT *
        FROM BU_ESG_WORK.PROCESS_PROPERTY_REL 
        WHERE DATA_DATE_PART IN (SELECT MAX (DATA_DATE_PART) FROM BU_ESG_WORK.PROCESS_PROPERTY_REL WHERE DATA_DATE_PART <= '${REF_DATE}')
        -- AND CONCAT(PROPERTY_BRANCH_CODE, PROPERTY_CONTRACT_ID, PROPERTY_REFERENCE_CODE) = '000000770100000000000000000000'
        -- AND CONCAT(PROCESS_BANK_ID,PROCESS_BRANCH_CODE, PROCESS_ACCOUNT_ID, PROCESS_REFERENCE_CODE) IN ('89000000000000000000000069614423',
        -- '89000000000000000000000069614421', '89000000000000000000000069614424', '89000000000000000000000069614422')
        ) PPR
    
    ON SUBSTR(LT006.PRTCTN_ID,3,4) = PPR.PROPERTY_BRANCH_CODE
    AND SUBSTR(LT006.PRTCTN_ID,8,11) = PPR.PROPERTY_CONTRACT_ID
    AND SUBSTR(LT006.PRTCTN_ID,20,15) = PPR.PROPERTY_REFERENCE_CODE
    AND RPR.PROCESS_BANK_ID = PPR.PROCESS_BANK_ID
    AND RPR.PROCESS_BRANCH_CODE = PPR.PROCESS_BRANCH_CODE
    AND RPR.PROCESS_ACCOUNT_ID = PPR.PROCESS_ACCOUNT_ID
    AND RPR.PROCESS_REFERENCE_CODE = PPR.PROCESS_REFERENCE_CODE
    
    -- WHERE COLLATERAL_PROPERTY_IND IS NOT NULL
    
    -- WHERE INSTRMNT_ID = '8200000000696144200Y430000471836'
    -- AND PRTCTN_ID = 'I_0000_16544930000_000000000000000'
    
    ) X
;
