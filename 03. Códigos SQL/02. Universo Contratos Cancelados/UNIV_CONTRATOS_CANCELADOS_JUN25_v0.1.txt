-------------------- DEFINIÇÃO DO UNIVERSO DE CONTRATOS CANCELADOS --------------------

-- DATA: 2025-06-30
-- DATA ANTERIOR: 2025-05-31

-- NOTA: TER EM ATENÇÃO QUE PARA OS CASOS DE FACTORING E CONFIRMING NÃO PODEMOS OLHAR PARA AS FATURAS NO MOMENTO EM QUE VALIDAMOS SE O CONTRATO É CANCELADO OU NÃO, OU SEJA, TEMOS DE OLHAR PARA O CONTRATO
-- ATRAVÉS DA AGREGAÇÃO DOS CAMOS CEMPCTA, CKBALCAO E CKNUMCTA (NECESSÁRIO VOLTAR A VALIDAR COM A EQUIPA DE CDG, POIS ISTO NÃO FICOU IMPLEMENTADO PELA EQUIPA ANTERIOR)

-- AUXILIAR 1: CONTRATOS DE CDG QUE FORAM CANCELADOS AO LONGO DO ANO, OU SEJA, PARA CADA MÊS DO ANO DE REPORTE ANALISAR OS CONTRATOS QUE NÃO TINHAM SALDO PONTUAL MAS TINHAM RENTABILIDADE

-- SELEÇÃO DAS OPERAÇÕES COM RENTABILIDADE E QUE NÃO TÊM POSIÇÃO DE FIM DE MÊS (C007=0); E NÃO TÊM CHAVE FINREP PARA O MÊS DE REPORTE

(
DROP TABLE IF EXISTS BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX1
)
;

(
CREATE TABLE BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX1 AS

SELECT DISTINCT KT_MIS.*, TIPOLOGIA
FROM 
    (
    SELECT * 
    FROM CD_CAPTOOLS.KT_CHAVES_MIS 
    WHERE REF_DATE='${DATE}'
    ) KT_MIS
    
INNER JOIN 
    (
    SELECT DISTINCT
        CEMPCTA, 
        CKBALCAO, 
        CKNUMCTA, 
        ZDEPOSIT,
        CKPRODMI,
        ZSEQDIV,
        CHCRUZA,
        CKMETAMIS,
        ZCLIENTE,
        CASE
            WHEN CKMETAMIS LIKE 'AI08%' THEN 'CARTÕES'
            ELSE 'OUTROS'
        END AS TIPOLOGIA
    
    FROM CD_CAPTOOLS.CT209_RENT_KPM 
    WHERE REF_DATE='${DATE}' 
    AND C001<>0 
    AND C007=0 
    AND (ABS(C001) + ABS(C007) + ABS(C134) + ABS(C224) + ABS(C298) + ABS(C299) + ABS(C300) + ABS(AC310) + ABS(M009)) <> 0
    AND ORIGEM NOT IN ('CAPITAL','CONTABILIDADE')
    -- AND CONCAT(CEMPCTA, CKBALCAO, CKNUMCTA, ZDEPOSIT) = '31003192196490011607616000000001'
    ) CT209

ON KT_MIS.CEMPCTA=CT209.CEMPCTA
AND KT_MIS.CKBALCAO=CT209.CKBALCAO
AND KT_MIS.CKNUMCTA=CT209.CKNUMCTA
AND KT_MIS.ZDEPOSIT_MIS=CT209.ZDEPOSIT

INNER JOIN 
    (
	SELECT DISTINCT METAPRODUTO 
	FROM CD_CAPTOOLS.CT831_CLASS_META
	WHERE REF_DATE = '${DATE_CT831}'  -- NAS TABELAS DE PARAMETRIZAÇÃO TEMOS DE VER PELA DATA EM QUE ESTAMOS A FAZER A SIMULAÇÃO
	AND AMBITO = 'GD'
	AND UPPER(TIPO_CONTA_LOCAL) NOT LIKE "%PASSIVO%"
	AND TRIM(METAPRODUTO) <> ''
    ) CT831
    
ON CT209.CKMETAMIS <=> CT831.METAPRODUTO

LEFT JOIN 
    (
    SELECT * 
    FROM CD_CAPTOOLS.KT_CHAVES_FINREP 
    WHERE REF_DATE = '${DATE}'
    -- WHERE CONCAT(CEMPRESA, cbalcao, cnumecta, zdeposit) = '89000020140001123001520230387637'
    ) FR_ATU

ON KT_MIS.CEMPRESA=FR_ATU.CEMPRESA
AND KT_MIS.CBALCAO=FR_ATU.CBALCAO
AND KT_MIS.CNUMECTA=FR_ATU.CNUMECTA
AND KT_MIS.ZDEPOSIT=FR_ATU.ZDEPOSIT

WHERE FR_ATU.CEMPRESA IS NULL
)
;

-- AUXILIAR 2: IDENTIFICAÇÃO DO UNIVERSO CANCELADO QUE SE ENCONTRAVA VIVO NO MÊS ANTERIOR (EXISTE NO TRADUTOR DO MÊS ANTERIOR),
-- E CRUZAMENTO COM A LT001 PARA NÃO TRAZER CHAVES QUE JÁ SE ENCONTRAM NO UNIVERSO DA CONTABILIDADE
(
DROP TABLE IF EXISTS BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX2
)
;

(
CREATE TABLE BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX2  AS 

SELECT DISTINCT 
    UNI.CEMPCTA, 
    UNI.CKBALCAO, 
    UNI.CKNUMCTA, 
    UNI.ZDEPOSIT_MIS, 
    KT_FR.CEMPRESA_FR, 
    KT_FR.CBALCAO_FR, 
    KT_FR.CNUMECTA_FR, 
    KT_FR.ZDEPOSIT_FR,
    '${DATE}' AS REF_DATE,
    'GD' AS AMBITO,
    'ESTR' AS SEGMENTO, 
    -- '1' AS ZCLIENTE
    TIPOLOGIA

FROM BU_CAPTOOLS_WORK.UNIV_CANCELADOS_jun25_AUX1 UNI

INNER JOIN 
    (
    SELECT * 
    FROM CD_CAPTOOLS.KT_CHAVES_FINREP 
    WHERE REF_DATE='${DATE_ANT}'
    ) KT_FR
    
ON UNI.CEMPRESA=KT_FR.CEMPRESA
AND UNI.CBALCAO=KT_FR.CBALCAO
AND UNI.CNUMECTA=KT_FR.CNUMECTA
AND UNI.ZDEPOSIT=KT_FR.ZDEPOSIT

LEFT JOIN 
    (
    SELECT *
    FROM BU_LOANTAPE_WORK.LT001_UNIV_OPER -- 13/MAI: ALTEREI A CD_LOAN_TAPES_BCE PELA BU POIS A CD NAO TINHA VALORES PARA FEV25
    WHERE REF_DATE ='${DATE}'
    AND AMBITO = 'PM'
    AND NOME_PERIMETRO = 'Individual IDComb'
    AND SEGMENTO = 'ESTR' 
    -- AND ENTIDADE = '00100'
    -- AND CEMPRESA IN ('80', '31') -- 31: SANTANDER & 80: TOTTA IRELAND 
    ) LT001
    
ON LT001.CEMPRESA = KT_FR.CEMPRESA_FR
AND LT001.CBALCAO = KT_FR.CBALCAO_FR
AND LT001.CNUMECTA = KT_FR.CNUMECTA_FR
AND LT001.ZDEPOSIT = KT_FR.ZDEPOSIT_FR

WHERE LT001.CEMPRESA IS NULL
)
;

-- AUXILIAR 3: IDENTIFICAÇÃO DA CHAVE FINREP PARA O UNIVERSO CANCELADO QUE NÃO TINHA SIDO APANHADO NO PROCESSO ANTERIOR,
-- E CRUZAMENTO COM A LT001 PARA NÃO TRAZER CHAVES QUE JÁ SE ENCONTRAM NO UNIVERSO DA CONTABILIDADE

(
DROP TABLE IF EXISTS BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX3
)
;

(
CREATE TABLE BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX3  AS 

SELECT DISTINCT 
    CEMPCTA, 
    CKBALCAO, 
    CKNUMCTA, 
    ZDEPOSIT_MIS, 
    CEMPRESA_FR, 
    CBALCAO_FR, 
    CNUMECTA_FR, 
    ZDEPOSIT_FR,
    '${DATE}' AS REF_DATE,
    'GD' AS AMBITO,
    'ESTR' AS SEGMENTO, 
    -- '2' AS ZCLIENTE
    TIPOLOGIA

FROM 
    (
    SELECT TIPOLOGIA, CEMPCTA, CKBALCAO, CKNUMCTA, ZDEPOSIT_MIS, CEMPRESA_FR, CBALCAO_FR, CNUMECTA_FR, ZDEPOSIT_FR
    FROM 
        (
        SELECT UNI.TIPOLOGIA, UNI.CEMPCTA, UNI.CKBALCAO, UNI.CKNUMCTA, UNI.ZDEPOSIT_MIS, KT_FR.CEMPRESA_FR, KT_FR.CBALCAO_FR, KT_FR.CNUMECTA_FR, KT_FR.ZDEPOSIT_FR,
        DENSE_RANK() OVER (PARTITION BY UNI.CEMPCTA, UNI.CKBALCAO, UNI.CKNUMCTA, UNI.ZDEPOSIT_MIS ORDER BY KT_FR.REF_DATE DESC) NROW
        
        FROM BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX1 UNI
        
        INNER JOIN 
            (
            SELECT *
            FROM CD_CAPTOOLS.KT_CHAVES_FINREP 
            WHERE REF_DATE<'${DATE_ANT}' 
            ) KT_FR
            
        ON UNI.CEMPRESA=KT_FR.CEMPRESA
        AND UNI.CBALCAO=KT_FR.CBALCAO
        AND UNI.CNUMECTA=KT_FR.CNUMECTA
        AND UNI.ZDEPOSIT=KT_FR.ZDEPOSIT
        
        LEFT JOIN BU_CAPTOOLS_WORK.UNIV_CANCELADOS_jun25_AUX2 UNI_MORTO
        
        ON UNI.CEMPCTA=UNI_MORTO.CEMPCTA
        AND UNI.CKBALCAO=UNI_MORTO.CKBALCAO
        AND UNI.CKNUMCTA=UNI_MORTO.CKNUMCTA
        AND UNI.ZDEPOSIT_MIS=UNI_MORTO.ZDEPOSIT_MIS
        
        WHERE  UNI_MORTO.CEMPCTA IS NULL
        ) AUX
    
    WHERE AUX.NROW=1
    ) BASE

LEFT JOIN 
    (
    SELECT *
    FROM BU_LOANTAPE_WORK.LT001_UNIV_OPER
    WHERE REF_DATE ='${DATE}'
    AND AMBITO = 'PM'
    AND NOME_PERIMETRO = 'Individual IDComb'
    AND SEGMENTO = 'ESTR' 
    -- AND ENTIDADE = '00100'
    -- AND CEMPRESA IN ('80', '31') -- 31: SANTANDER & 80: TOTTA IRELAND 
    ) LT001

ON LT001.CEMPRESA = BASE.CEMPRESA_FR
AND LT001.CBALCAO = BASE.CBALCAO_FR
AND LT001.CNUMECTA = BASE.CNUMECTA_FR
AND LT001.ZDEPOSIT = BASE.ZDEPOSIT_FR

WHERE LT001.CEMPRESA IS NULL
)
;

-- AUXILIAR 4: IDENTIFICAÇÃO DOS CONTRATOS CANCELADOS QUE NÃO FORAM APANHADOS EM NENHUM DOS OUTROS PASSOS E QUE POR ISSO NÃO APRESENTAM CHAVE FINREP, 
-- NESTE SENTIDO A CHAVE FINREP IRÁ CORRESPONDER À CHAVE MIS
-- E CRUZAMENTO COM A LT001 PARA NÃO TRAZER CHAVES QUE JÁ SE ENCONTRAM NO UNIVERSO DA CONTABILIDADE

(
DROP TABLE IF EXISTS BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX4
)
;

(
CREATE TABLE BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX4  AS 

SELECT DISTINCT 
    UNI.CEMPCTA, 
    UNI.CKBALCAO, 
    UNI.CKNUMCTA, 
    UNI.ZDEPOSIT_MIS,
    UNI.CEMPCTA AS CEMPRESA_FR, 
    UNI.CKBALCAO AS CBALCAO_FR, 
    UNI.CKNUMCTA AS CNUMECTA_FR, 
    UNI.ZDEPOSIT_MIS AS ZDEPOSIT_FR,
    '${DATE}' AS REF_DATE,
    'GD' AS AMBITO,
    'ESTR' AS SEGMENTO, 
    -- '3' AS ZCLIENTE
    UNI.TIPOLOGIA
    
FROM BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX1 UNI

LEFT JOIN BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX2  UNI_MORTO

ON UNI.CEMPCTA=UNI_MORTO.CEMPCTA
AND UNI.CKBALCAO=UNI_MORTO.CKBALCAO
AND UNI.CKNUMCTA=UNI_MORTO.CKNUMCTA
AND UNI.ZDEPOSIT_MIS=UNI_MORTO.ZDEPOSIT_MIS

LEFT JOIN BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX3 UNI_VIVO_MORTO

ON UNI.CEMPCTA=UNI_VIVO_MORTO.CEMPCTA
AND UNI.CKBALCAO=UNI_VIVO_MORTO.CKBALCAO
AND UNI.CKNUMCTA=UNI_VIVO_MORTO.CKNUMCTA
AND UNI.ZDEPOSIT_MIS=UNI_VIVO_MORTO.ZDEPOSIT_MIS

LEFT JOIN 
    ( 
    SELECT *
    FROM BU_LOANTAPE_WORK.LT001_UNIV_OPER
    WHERE REF_DATE ='${DATE}'
    AND AMBITO = 'PM'
    AND NOME_PERIMETRO = 'Individual IDComb'
    AND SEGMENTO = 'ESTR' 
    -- AND ENTIDADE = '00100'
    -- AND CEMPRESA IN ('80', '31') -- 31: SANTANDER & 80: TOTTA IRELAND 
    ) LT001
ON LT001.CEMPRESA = UNI.CEMPCTA
AND LT001.CBALCAO = UNI.CKBALCAO
AND LT001.CNUMECTA = UNI.CKNUMCTA
AND LT001.ZDEPOSIT = UNI.ZDEPOSIT_MIS

WHERE UNI_MORTO.CEMPCTA IS NULL AND UNI_VIVO_MORTO.CEMPCTA IS NULL AND LT001.CEMPRESA IS NULL
)
;

-- AGREGAÇÃO DAS VÁRIAS TABELAS AUXILIARES REFERENTES AO PROCESSO DE IDENTIFICAÇÃO DE CONTRATOS CANCELADOS

(
DROP TABLE IF EXISTS BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25
)
;

(
CREATE TABLE BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25  AS 

SELECT *
FROM BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX2 
UNION ALL
SELECT *
FROM BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX3
UNION ALL
SELECT *
FROM BU_CAPTOOLS_WORK.UNIV_CANCELADOS_JUN25_AUX4
)
;
